## Usage

```tsx
import {
	CouponApplicator,
	type ValidationResult,
} from '@/components/billingsdk/coupon-applicator';
```

```tsx
const handleApply = async (code: string): Promise<ValidationResult> => {
	// Validate coupon code (e.g., call your API)
	const response = await fetch(`/api/coupons/${code}`);
	const data = await response.json();

	return {
		isValid: data.valid,
		discount: data.valid ? data.discount : undefined,
		message: data.message,
	};
};

<CouponApplicator
	onApply={handleApply}
	currentPrice={99.99}
	onRemove={() => console.log('Coupon removed')}
/>;
```

## Props

| Prop           | Type                                          | Required | Description                                        |
| -------------- | --------------------------------------------- | -------- | -------------------------------------------------- |
| `onApply`      | `(code: string) => Promise<ValidationResult>` | ✅       | Async function to validate and apply coupon        |
| `currentPrice` | `number`                                      | ✅       | Current price for discount calculations            |
| `className`    | `string`                                      | ❌       | Additional CSS classes for styling                 |
| `theme`        | `"classic" \| "minimal"`                      | ❌       | Visual theme variant (default: "classic")          |
| `autoApply`    | `boolean`                                     | ❌       | Automatically apply on blur/Enter (default: false) |
| `onRemove`     | `() => void`                                  | ❌       | Callback when coupon is removed                    |

## ValidationResult

```ts
interface ValidationResult {
	isValid: boolean;
	discount?: {
		type: 'percentage' | 'fixed';
		value: number;
	};
	message?: string;
	newTotal?: number;
	savings?: number;
}
```
